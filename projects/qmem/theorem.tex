

\section{Proof of The Theorem}

  \begin{definition}[Simulation and Logical Errors]
For a quantum code $\mathcal{C}$ of length $n$ and a decoder for $\mathcal{C}$, denoted by $D$, we say that a Pauli operator $P \in \mathcal{P}^{\otimes n}$ is a \textit{logical error} with respect to $D$ if $DP$ does not equal the logical identity.

For quantum circuits $C$ and $\tilde{C}$, a noise channel $\mathcal{E}$, and $p \in (0,1)$, we say that $\tilde{C}$ \textit{simulates} $C$ with probability $1 - p$ if
    \begin{enumerate}
      \item If there is a mapping between the qubits of $C$ and the registers of $\tilde{C}$, along with a decoding procedure such that, at the end of running $\tilde{C}$, decoding the registers results in the same quantum states as those produced by running $C$.
      \item The probability induced by $\mathcal{E}$ that the deviation on one of the registers has a logical error is less than $p$.
    \end{enumerate}
  \end{definition}


 
  \begin{lemma}
  For any $d, d^{\prime}$, and a constant $p_{0} \in (0,1)$ that depends only on $d$. Consider a quantum circuit $C$ with depth $\mu$ and width $w$, and let $\mathcal{N}_{\mathcal{E}}$ be a local stochastic noise channel with a noise rate $p < p_{0}$. There is a Toric encoded circuit $\tilde{C}$ such that each of its quantum registers is encoded by an $n$-length, $(d, d^{\prime})$-Toric code. It simulates $C$ with probability: 
\end{lemma}

  \begin{proof}
    We construct $\tilde{C}$ as follows: for each input wire of $C$, we initialize the logical zero $\ket{\mathbf{0}}$ of the $(d,d^{\prime})$-Toric code. For each of the gates $\mathbf{H}$ and $\mathbf{S}$, we initialize the encoded $\ket{\mathbf{H}}$ and $\ket{\mathbf{S}}$ states, and for each $\mathbf{T}$, we initialize copies of both $\ket{\mathbf{T}}$ and $\ket{\mathbf{S}}$ states. The even layers of $\tilde{C}$ are sweep cycles, while on the odd layers, we set the realization of the logical gates. For $\mathbf{CX}$, we set the transversal realization, and for $\mathbf{H}$, $\mathbf{S}$, and $\mathbf{T}$, we use gate teleportation. Thus the width of $\tilde{C}$ expands to $ wn + 2\mu n$. For gate teleportation, we use the decoder in \Cref{ }, which runs at most at a depth of $n$. Thus, the depth of $\tilde{C}$ expands to $2(3\mu n)$.

    
The decoder successfully decodes any error that does not contain a deviation cluster larger than $\gamma n^{\frac{1}{d}}$. Thus, it remains to prove that with probability $1 - p$, no deviation cluster exceeds that size. To prove this, we use \Cref{claim:intersyn}, which states that the probability of there being a time $t$ with an error cluster larger than $\gamma n^{\frac{1}{d}}$ is bounded by the number of locations multiplied by the probability that step $\tau$ is the first to admit a deviation cluster of size $\gamma^{\prime} n^{\frac{1}{d}}$, maximized over $\tau$. This leads us to prove the following claim:

\begin{claim}
  There exist constants $c_{3}, c_{4}, c_{5}, c_{6}, \alpha \in \mathbb{R}^{+}$, such that for ant integer $s$ and for any step $\tau$ in the standard order of $\tilde{C}[\mathcal{E}]$, the probability that $\tau$ is the first step where there is a deviation cluster of size greater than $s$ is less than:
\begin{equation*}
  \begin{split}
    c_{3} (w+2\mu)\mu n^{2} \left( \alpha p^{c_{4}} \right)^{s} +  c_{5} ((w+2\mu)\mu n^{2})^{2} \left( \alpha p^{c_{6}} \right)^{\frac{1}{2d}n^{\frac{1}{d}}} 
  \end{split}
\end{equation*}
\end{claim}

\begin{proof}
Denote by $B_{\tau}$ the event where $\tau$ is the first step to have a deviation cluster of size greater than $\frac{1}{2}n^{\frac{d^{\prime}}{d}}$, and by $A_{\tau}$ the event where all explanations of deviation clusters up to and including step $\tau$ have fewer than $\frac{1}{d}n^{\frac{1}{d}}$ edges. Thus:
    \begin{equation*}
      \begin{split}
        \prbm{  B_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} } \le \prbm{  B_{\tau}  | A_{\tau}  }{ \sim \mathcal{N}_{\mathcal{E}} } + \prbm{  \bar{A}_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} }  
      \end{split}
    \end{equation*}
We begin by bounding the contribution of the right term. According to \Cref{coro:edgefaultratio}, there exists a constant $c_{1}$, depending only on $d$, such that the ratio of leaves to edges in the explanation is at least $c_{1}$. By \Cref{claim:big_subtree}, it follows that any such explanation has more than $\frac{1}{d}n^{\frac{1}{d}}$ edges containing a subtree with fewer than $\frac{1}{2d}n^{\frac{1}{d}}$ edges, with a leaves-to-edges ratio greater than $\frac{1}{2} c_{1}$. Using the union bound, we bound the probability that there is a vertex in the base graph that is a root of such a subtree.
  \begin{equation*}
    \begin{split}
      \prbm{  \bar{A}_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} }  \le  \bigcup_{v \in V_{o}, \tau^{\prime} \le \tau } \prbm{ & v \text{ is a root of subtree in } \mathcal{H}_{\tau^{\prime}}  \\ 
      &  \text{ with less than  } \frac{1}{2d}n^{\frac{1}{d}} \text{ edges and more than } \frac{1}{2}c_{1} \frac{1}{2d}n^{\frac{1}{d}} \text{ leaves}  }{ \sim \mathcal{N}_{\mathcal{E}} } 
    \end{split}
  \end{equation*}
    Now, since any two connected vertices are at a space distance of at most 
    \begin{equation*}
      \begin{split}
    \max\{\Size{\text{arrow}}, \Size{\text{fork}}\} \le d^{\prime} \le d
      \end{split}
    \end{equation*}
It follows that all the vertices in the subtree are at a space distance of at most $n^{\frac{1}{d}}$ from each other. Thus, no two of them have the same space coordinate modulo $n$. In particular, projecting the infinite plane back to the finite Toric sends each of the leaves of the subtree to a unique target, and therefore the number of faults that induce the explanation is at least:
    \begin{equation*}
      \begin{split}
\frac{1}{2^{d^{\prime}}} \frac{1}{4d} c_{1} n^{\frac{1}{d}}
      \end{split}
    \end{equation*}
    Plugging into the union bounds yields 
    \begin{equation*}
      \begin{split}
        \prbm{  \bar{A}_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} }  \le \tau \cdot 3 (w+2\mu)6\mu n^{2} \alpha^{ \frac{1}{2d} n^{\frac{1}{d}} } p^{ \frac{1}{2^{d^{\prime}}}\frac{1}{4d} c_{1} n^{\frac{1}{d}}} \le ((w+2\mu)6\mu n^{2})^{2} \left( \alpha p^{\frac{1}{2^{d^{\prime} + 1}} c_{1}} \right)^{\frac{1}{2d}n^{\frac{1}{d}}} 
      \end{split}
    \end{equation*}
    It remains to bound the term. By \Cref{claim:edgesin}, there is a constant $c_{2}$ depending only on $d$, such that any explanation of a slice of size $s$ has at least $c_{2}s$ edges. Therefore, it suffices to bound the probability that the 'analysis Toom's contour', $\mathcal{H}_{\tau}$ admits an explanation with more than $c_{2}s$ edges. Since the event is conditioned on the event $A_{\tau}$, no explanation has more than $\frac{1}{d}n^{\frac{1}{d}}$ edges. Thus, the same argument as above shows that the projection of each explanation tree into the finite toric sends its leaves to unique targets. Hence, we obtain:
    \begin{equation*}
      \begin{split}
        \prbm{  B_{\tau}  | A_{\tau}  }{ \sim \mathcal{N}_{\mathcal{E}} } & \le \bigcup_{ v \in V_{o}, x \ge c_{2}s } \prbm{ v \text{ is a root of explanation with number of edges } x  | A_{\tau}  }{ \sim \mathcal{N}_{\mathcal{E}} } \\
        & \le \sum_{x = c_{2}s }^{ \infty } (w+2\mu)6\mu n^{2} \left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)^{x}  = (w+2\mu)6\mu n^{2} \frac{\left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)^{s}}{ 1 -  \left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)} 
      \end{split}
    \end{equation*}
Combing the bounds we have:
\begin{equation*}
  \begin{split}
\prbm{  B_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} } \le (w+2\mu)6\mu n^{2} \frac{\left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)^{c_{2}s}}{ 1 -  \left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)} +  ((w+2\mu)6\mu n^{2})^{2} \left( \alpha p^{\frac{1}{2^{d^{\prime} + 1}} c_{1}} \right)^{\frac{1}{2d}n^{\frac{1}{d}}} 
  \end{split}
\end{equation*}
\end{proof}
Now, combined with \Cref{claim:intersyn}, with a probability greater than:
\begin{equation*}
  \begin{split}
    1 -   \Theta \left(((w+2\mu)\mu n^{2})^{3} \right) \left( \alpha p^{c_{7}} \right)^{ \gamma^{\prime} n^{\frac{1}{d}}} 
  \end{split}
\end{equation*}
No error cluster has a size greater than $ \gamma n^{\frac{1}{d}} $, so the decoder successfully restores the logical state; in other words, no logical error occurs.
\end{proof}
