\begin{claim}
Assume that until step $\tau$, there was no deviation cluster of size greater than $d$. Let $\hat{K}$ be a spanned slice induced by a deviation of a cluster state at step $\tau$, and let $\omega$ be an explanation for $\hat{K}$ such that the number of vertices in $\omega$ is greater than $\beta n$. Then there is a subgraph (subtree) $\omega^{\prime}$ of $\omega$ such that $\omega^{\prime}$ has fewer than $\frac{\beta}{2}n$ edges and at least $\frac{\beta}{4\gamma} n$ leaves.
\end{claim}







\begin{proof}
Let $\omega$ be an explanation of $\hat{K}$ with more than $\beta n$ vertices. Since there is no deviation cluster of size greater than $d$ until step $\tau$, we have that every vertex of $\omega$ is a spanned slice induced by a boundary of a deviation cluster of size at most $d$. Thus, the image of $\omega$ over the infinite plane is decomposed into finite assignments, and $\omega$ satisfies the shrink property in \Cref{claim:shrinkslice}, and by \Cref{claim:edgesin}, the ratio of edges to leaves of $\omega$ is at most $\gamma$.
Observe that a spanning tree of $\omega$ has a ratio of edges to leaves that is at most $\omega$'s ratio, and the number of edges equals $\omega$'s vertices minus 1. Therefore, by applying \Cref{claim:big_subtree} on $\omega$'s spanning tree, $\omega$ has a subtree with less than $\frac{\beta}{2}n$ edges and at least $\frac{\beta}{4\gamma} n$ leaves. 
  \end{proof}

 
\begin{claim}
Let $C$ be a Toric encoded circuit such each of its quantum register is ecnoded by $n$-length , $(d,d^{\prime})$-Toric code, and let $\mathcal{N}_{\mathcal{E}}$ be a local stochastic noise channel, with a noise rate $p$. Then, for any step $\tau$ in the standard order of $C[\mathcal{E}]$, the probability that $\tau$ is the first step such there is a deivation cluster at size greather than $ \frac{1}{2} n^{\frac{d^{\prime}}{d}} $ is less than $  $. 
\end{claim}

  \begin{proof}
    
    Denote by $B_{\tau}$ the event that $\tau$ is the first step to have a diveation cluster at size greater than $\frac{1}{2}n^{\frac{d^{\prime}}{d}}$, and by $A_{\tau}$ the event that all the explantations of deviation clusters until (and including) step $\tau$ have less than $\frac{1}{d}n^{\frac{1}{d}}$ edges. Thus:  
    \begin{equation*}
      \begin{split}
        \prbm{  B_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} } \le \prbm{  B_{\tau}  | A_{\tau}  }{ \sim \mathcal{N}_{\mathcal{E}} } + \prbm{  \bar{A}_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} }  
      \end{split}
    \end{equation*}
We begin by bounding the contribution of the right term. According to \Cref{coro:edgefaultratio}, there exists a constant $c_{1}$, depending only on $d$, such that the ratio of leaves to edges in the explanation is at least $c_{1}$. By \Cref{claim:big_subtree}, it follows that any such explanation has more than $\frac{1}{d}n^{\frac{1}{d}}$ edges containing a subtree with fewer than $\frac{1}{2d}n^{\frac{1}{d}}$ edges, with a leaves-to-edges ratio greater than $\frac{1}{2} c_{1}$. Using the union bound, we bound the probability that there is a vertex in the base graph that is a root of such a subtree.
  \begin{equation*}
    \begin{split}
      \prbm{  \bar{A}_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} }  \le  \bigcup_{v \in V_{o}, \tau^{\prime} \le \tau } \prbm{ & v \text{ is a root of subtree in } \mathcal{H}_{\tau^{\prime}}  \\ 
      &  \text{ with less than  } \frac{1}{2d}n^{\frac{1}{d}} \text{ edges and more than } \frac{1}{2}c_{1} \frac{1}{2d}n^{\frac{1}{d}} \text{ leaves}  }{ \sim \mathcal{N}_{\mathcal{E}} } 
    \end{split}
  \end{equation*}
    Now, since any two connected vertices are at a space distance of at most 
    \begin{equation*}
      \begin{split}
    \max\{\Size{\text{arrow}}, \Size{\text{fork}}\} \le d^{\prime} \le d
      \end{split}
    \end{equation*}
It follows that all the vertices in the subtree are at a space distance of at most $n^{\frac{1}{d}}$ from each other. Thus, no two of them have the same space coordinate modulo $n$. In particular, projecting the infinite plane back to the finite toric space sends each of the leaves of the subtree to a unique target, and therefore the number of faults that induce the explanation is at least:
    \begin{equation*}
      \begin{split}
\frac{1}{2^{d^{\prime}}} \frac{1}{4d} c_{1} n^{\frac{1}{d}}
      \end{split}
    \end{equation*}
    Plugging into the union bounds yields 
    \begin{equation*}
      \begin{split}
        \prbm{  \bar{A}_{\tau}    }{ \sim \mathcal{N}_{\mathcal{E}} }  \le \tau \cdot 3 nwd \alpha^{ \frac{1}{2d} n^{\frac{1}{d}} } p^{ \frac{1}{2^{d^{\prime}}}\frac{1}{4d} c_{1} n^{\frac{1}{d}}} \le (3 nwd)^{2} \left( \alpha p^{\frac{1}{2^{d^{\prime} + 1}} c_{1}} \right)^{\frac{1}{2d}n^{\frac{1}{d}}} 
      \end{split}
    \end{equation*}
    It remains to bound the term. By \Cref{claim:edgesin}, there is a constant $c_{2}$ depending only on $d$, such that any explanation of a slice of size $s$ has at least $c_{2}s$ edges. Therefore, it suffices to bound the probability that the 'analysis Toom's contour', $\mathcal{H}_{\tau}$ admits an explanation with more than $c_{2}s$ edges. Since the event is conditioned on the event $A_{\tau}$, no explanation has more than $\frac{1}{d}n^{\frac{1}{d}}$ edges. Thus, the same argument as above shows that the projection of each explanation tree into the finite toric sends its leaves to unique targets. Hence, we obtain:
    \begin{equation*}
      \begin{split}
        \prbm{  B_{\tau}  | A_{\tau}  }{ \sim \mathcal{N}_{\mathcal{E}} } & \le \bigcup_{ v \in V_{o}, x \ge c_{2}\frac{1}{2}n^{d} } \prbm{ v \text{ is a root of explanation with number of edges } x  | A_{\tau}  }{ \sim \mathcal{N}_{\mathcal{E}} } \\
        & \le \sum_{x = c_{2}s }^{ \infty } nwd \left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)^{x} \\ 
        & = nwd \frac{\left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)^{c_{2}s}}{ 1 -  \left( \alpha p^{\frac{1}{2^{d^{\prime} }} c_{2}} \right)} 
      \end{split}
    \end{equation*}
Because any $\omega$ with more than $\beta n$ vertices implies the existence of a subtree with size less than $\frac{\beta}{2}n$, it's enough to bound the probability of having such a subtree. Considering that the trees are invariant to shifting by $n \cdot \mathbf{1}_{i}$, we have that it's sufficient to show that there are no such trees which are rooted in the finite cube $\mathbb{Z}_{n}^{d} \times \mathbb{Z}_{t}$. Finally, we use \cref{claim:counting_trees} to bound the number of such trees given a fixed root. Overall, we get:
  \begin{equation*}
    \begin{split}
      \prb{\text{No } \omega \text{s with more than  } \beta n \text{ vertices} } \le n^{d} t  \left( \xi^{2}  p^{\frac{\beta}{4\gamma}} \right)^{n}   
    \end{split}
  \end{equation*}
\end{proof}
