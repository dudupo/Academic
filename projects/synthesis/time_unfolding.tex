Informally, a quantum circuit is a placement of a collection of quantum gates, taken from a finite set, in different space-time locations. Each space-time location, associated with a gate, encodes the time of the application and the qubits in the support of the gate. Formally

\begin{definition}[Quantum Circuit]
  A space-time location $l$ is a tuple that encodes in its first coordinate time, and in its second coordinate a subset of integers referring to a support of (q)bits. 
  For a collection of quantum gates $\mathcal{G} = \{ g \colon L(\mathcal{H}_{2}^{\otimes \Delta} ) \rightarrow L(\mathcal{H}_{2}^{\otimes \Delta} ) \}$, and integers $w,d \in \mathbb{N}$ a quantum circuit $C$, with depth $d$ and width $w$, is a collection of tuples $\{ u_{i} = (g_{i},l_{i}) \}_{i=1}$, where $u_{i}$ are the gates of $C$ and each $u_{i}$ is composed of a gate $g_{i} \in \mathcal{G}$ and a space-time location $l_{i} \in \mathbb{Z}_{d} \times \mathbb{Z}_{w}^{\Delta}$. The gates of $C$ have to satisfy the constraint that for every two gates with the same time coordinate, there cannot be a non-empty intersection in their space coordinate, namely for $u_{i}=(g_{i},l_{i}),u_{j}=(g_{j},l_{j}) \in C$, the equality $l_{i,1} = l_{j,1}$ implies $l_{i,2} \cap l_{j,2} = \emptyset$. 
  We define the $t$-cut of $C$ to be $C|_{t} \colonequals \{ u_{i} = (g_{i},l_{i}) \colon u_{i} \in C \text{ and } l_{i,1} = t \}$.
\end{definition}
To define the running of a quantum circuit $C$ on a given (mixed) quantum state, we will need to have the notion of the directed acyclic graph associated with $C$.

\begin{definition}[Associated Computation DAG]
  Let $C = \{u_{i}\}_{i=1}$ be a quantum circuit with depth $d$ and width $w$. We define the associated computation DAG of $C$, denoted by $\pi_{C}$, to be the acyclic directed graph $\pi_{C} = (\mathcal{U}, E)$ as follows:
  \begin{enumerate}
    \item The vertex set $\mathcal{U}$ is the gate set of $C$, namely $\mathcal{U} = \{u_{i}\}_{i=1}$.
    \item For $u_{i}= (g_{i},l_{i}), u_{j}= (g_{j},l_{j})$, There is an edge $\{u_{i},u_{j}\} \in E$ if and only if:
      \begin{equation*}
        \begin{split}
          j \in \arg \min_{k} \big\{ l_{k,1} : (g_{k},l_{k})\in \mathcal{U}, l_{k,1} > l_{i,1} \text{ and } l_{k,2}\cap l_{i,2} \neq \emptyset  \big\}
        \end{split}
      \end{equation*}
Note that there might be multiple indices that get the minimum, so the out degree might be greater than one.
  \end{enumerate}
  In addition, suppose that $J = \{ j_{1}, j_{2}, \ldots, j_{k} \} \subset [|C|]$ is an order of the indices of the gates in $C$ according to the topological order induced by $\pi_{C}$. Then, we say that $J$ is an order of indices that agrees with $\pi_{C}$ and define the series of circuit $C_{J} = \{ C_{J,i} \}_{i=1}$ as follows:
  \begin{enumerate}
    \item The first element: $C_{J,1} = \{ u_{j_{1}} \}$ 
    \item Then, $C_{J,i+1} = C_{J,i} \cup \{ u_{j_{i}} \}$. We name $C_{J,i}$ the partial computation circuit until step $i$.  
  \end{enumerate}
\end{definition}

\begin{definition}
  Let $\rho$ be a density matrix over $w$ qubits, and let $C = \{u_{i}\}_{i=1}$ be a quantum circuit with depth $d$ and width $w$. For any density matrix $\sigma$, quantum gate $g \in \mathcal{G}$, and location $l$, the application of the gate-location tuple $(g,l)$ on $\sigma$, denoted by $(g,l) \circ \sigma$, is the application of $g$ on $\sigma$ where wires are ordered such that the first qubit in the input to $g$ is $l_{2,1}$, the second is $l_{2,2}$, and so on. The result of the application of $C$ on $\rho$, denoted by $C \circ \rho$, is given by iterative applications of the gates $\{u_{i}\}_{i=1}$ on $\rho$, according to a topological order induced by $\pi_{C}$.

  We say that $\rho_{1}, \rho_{2}, \ldots, \rho_{k}$ is a computation prefix if $\rho_{1} = \rho$ and $\rho_{i+1} = u^\prime_{i} \circ \rho_{i}$, where $\{u^{\prime}_{i}\}_{i=1}$ is a prefix of a topological ordering of $\{u_{i}\}_{i=1}$. If $\{u^{\prime}_{i}\}_{i=1}$ contains all the gates in $\{u_{i}\}_{i=1}$, then we call $\rho_{1}, \rho_{2}, \ldots, \rho_{k}$ a running.
\end{definition}

\begin{definition}[$C$ Subjected to $\mathcal{E}$ at frequency $r$]
  Let $C = \{ u_{i} \}_{i=1}$ be a quantum circuit with depth $d$ and width $w$. For a set of fault locations $\mathcal{E} \subset [d] \times [w]$, and a number $r \in (0,1)$ satisfies $r^{-1} \in \mathbb{N}$, we define the quantum circuit \textit{$C$ subjected to $\mathcal{E}$ at frequency $r$}, denoted by $C[\mathcal{E}]_{r} = \{ u^{\prime}_{i} \}_{i=1}$, to be the quantum circuit that is the result of the following map:
  \begin{equation*}
    \begin{split}
      u_{i}=(g_{i},(l_{i,1}, l_{i,2})) & =
      \begin{cases}
        (g_{i},(  l_{i,1} + \inb{ \lfloor r  l_{i,1} \rfloor + 1 }, l_{i,2})) & u_{i} \in C \\ 
        (g_{i},( \inb{r^{-1}} l_{i,1}, l_{i,2})) & u_{i} \in \mathcal{E}
      \end{cases}
    \end{split}
  \end{equation*} 
Where $r$ is not mentioned, we mean $r = \frac{1}{2}$, and the result of the subjection is the circuit whose gates on the odd time coordinates are the gates of $C$, and whose gates on the even time coordinates are the gates of $\mathcal{E}$. We will denote the associated computation DAG of $C$ affected by $\mathcal{E}$ as $\comDAG$.
\end{definition}

\begin{definition}[Faults Propagation to a Step $\tau$]
  For a quantum circuit $C=\{u_{i}\}_{i=1}$, with depth $d$, an order $J = \{ j_{1}, j_{2}, \ldots, j_{k} \} \subset [|C|]$ that agrees with $\pi_{C}$, and gate $u_{i}$ in $C$, a propagation of a gate $u_{i} \in C$ to step $\tau$ is the quantum operator $X\colon L(\mathcal{H}_{2}^{w}) \rightarrow L(\mathcal{H}_{2}^{w})$ obtained by the follow process: 
  \begin{enumerate}
    \item First, check if $i$ precedes $\tau$ in $J$. If not, then return $X = I$. Otherwise, denote $j_{l} = i$, and look for an operator $X_{1}$ such that $u_{j_{l}}u_{j_{l+1}} = u_{j_{l+1}}X_{1}$.
    \item Then, for $k \le \tau - l$, look for an operator $X_{k} \colon L(\mathcal{H}_{2}^{w}) \rightarrow L(\mathcal{H}_{2}^{w})$, such that  $X_{k-1}u_{j_{l+1}} =  u_{j_{l+1}}X_{k}$.
    \item At the final stage of the process, namely where $k = \tau - l$, we define $X \colonequals X_{k}$ to be the propagation of $u_{j}$ to step $\tau$.
  \end{enumerate}
  For a quantum circuit $C$, a set of faults $\mathcal{E} = \{ e_{i} \}_{i=1}$, an order $J \subset [|C[\mathcal{E}]|]$ that agrees with $\pi_{C[\mathcal{E}]}$, and a step $\tau \in J$, denote by $X_{e_{i}}^{\tau}$ the propagation of $e_{i} \in \mathcal{E}$ in $C[\mathcal{E}]$ to step $\tau$. Let $j^{\prime}_{1}, j^{\prime}_{2}, \ldots, j^{\prime}_{l}$ be the indices of the faults in $\mathcal{E}$ such that their corresponding positions in $J$ precede $\tau$, ordered according to the reverse topological order induced by $\pi_{\mathcal{E}}$. The propagated error to step $\tau$ is defined by
  \begin{equation*}
    \begin{split}
      X^{\tau} \colonequals \prod_{i} X_{e_{j^{\prime}_{i}}}^{\tau} 
    \end{split}
  \end{equation*}
\end{definition}

\begin{definition}[ $C \sim C^{\prime}$]
We define the equivalence relation over quantum circuits $\sim$ to be satisfied if any of them can be obtained from the other by propagating gates. Namely, for quantum circuits $C, C^{\prime}$, we have $C \sim C^{\prime}$ if there is a subset of gates $\{ u_{i} \}_{i=1} \subset C$ and subsets of turns $\{ \tau_{i} \}_{i=1}$ such that $C^{\prime}$ is the result of propagating $u_{i}$ to turn $\tau_{i}$ in $C$ according to the order indexed by $i$.
\end{definition}

\begin{definition}[Deviation]
  \label{def:deviation}
  For a quantum circuit $C$, a set of faults $\mathcal{E}$, and step $\tau$, let $X^{\tau}$ be the propagated error at step $\tau$. We define the deviation of $C[\mathcal{E}]$ from $C$ at step $\tau$ by the set of indices on which $X^{\tau}$ acts non-trivially. 
\end{definition}
\begin{definition}
  Using the same notations as in \Cref{def:deviation}, for a set of unitary operators $\mathcal{S} = \{s_{i} \colon \mathcal{H}_{2}^{w} \rightarrow \mathcal{H}_{2}^{w} \}_{i=1}$, we define the propagated error up to stabilizers in $\mathcal{S}$ as $X^{\tau}[S] \colonequals \min_{s \in \mathcal{S}} \{ w(sX) \}$. Then, we define the deviation of $C[\mathcal{E}]$ from $C$ at step $\tau$ up to stabilizers in $\mathcal{S}$ to be the subset of coordinates on which $X^{\tau}[S]$ acts non-trivially.
\end{definition}

The following, extends the notion of Tanner graphs to the context of quantum circuits. 

\begin{definition}[Tanner Graph of a Register]
  For a quantum circuit $C$, a register $R$ is a subset of qubits in $C$. The \textit{Tanner graph} at step $\tau$ of register $R$, denoted by $\mathcal{T}_{R, \tau}$, is the graph defined as follows:
  \begin{enumerate}
    \item If there is a code $\mathcal{C}$, such that at turn $\tau$, the qubits in $C$ are an encoding of $\mathcal{C}$, then $\mathcal{T}_{R, \tau}$ is the Tanner graph of $\mathcal{C}$.
    \item Otherwise, $\mathcal{T}_{R, \tau}$ is the empty graph.
  \end{enumerate}
Where we consider a range of turns in which $R$ is encoded in the same code, and the turn is clear from the context, we will be brief by omitting the turn subscript.
\end{definition}
\begin{definition}[Generalized Tanner Graph of a Circuit]
For a quantum circuit $C$, and registers $a$ and $b$ in $C$, the \textit{generalized Tanner graph} of the register obtained by taking their union $a \cup b$ is $\mathcal{T}_{a \cup b} \colonequals \mathcal{T}_{a} \cup \mathcal{T}_{b}$. For a partition of the qubits into registers $\{R_{i}\}_{i=1}$, the \textit{generalized Tanner graph} of a quantum circuit $C$, at step $\tau$, according to the partition $\{R_{i}\}_{i=1}$, is the graph obtained by:
  \begin{equation*}
    \begin{split}
      \mathcal{T}_{C, \{R_{i}\}_{i=1}, \tau } \colonequals \bigcup_{i=1} \mathcal{T}_{R_{i}, \tau}
    \end{split}
  \end{equation*}
When the partition is clear, we will brief and use $\mathcal{T}_{C}$ to define the generalized Tanner graph of $C$.
\end{definition}

\begin{definition}[Clustered Deviation]
For a quantum circuit $C$, a register $R$ of $C$, and a set of faults $\mathcal{E}$, let $\mathcal{T}_{R}$ be the Tanner graph of $R$, and let $\mathcal{D}_{\tau}$ be the deviation of $C[\mathcal{E}]$ from $C$ on $R$ at step $\tau$. Denote by $\mathcal{T}_{R}|_{\mathcal{D}_{\tau}}$ the subgraph of $\mathcal{T}_{R}$ induced by the restriction of the left vertices (bits) to the indices in $\mathcal{D}_{\tau}$. Let us abuse and extend the connection notion to Tanner graphs by defining that two left vertices are connected if and only if they share a right neighbor. We define the deviation clusters at step $\tau$ to be the connected components of $\mathcal{T}_{R}|_{\mathcal{D}_{\tau}}$. The size of each deviation cluster is the number of right vertices it contains.
\end{definition}

\begin{corollary}[Intermediate Value Theorem]
  \label{claim:inter}
  Let $\alpha = 1 / (\Delta\Delta_{1}\Delta_{2})$, and let $x$ be an integer. Assume that at the initial step $\tau_{o}$, there is no deviation cluster of size more than $\alpha x$, and that at a step $\tau_{1} \in J$, such that $\tau_{1} > \tau_{o}$, there is a deviation cluster $Y^{(\tau_{1})}$ of size $|Y^{(\tau_{1})}| > x$. Then there is a step $\tau_{\star} \in J$ such that $\tau_{o} < \tau_{\star} < \tau_{1}$ for which there is a deviation cluster $Z$ at step $\tau_{\star}$, satisfying that the size of $Z$ is in the range $(\alpha x, x)$.
\end{corollary}
\begin{proof}
Assume through contradiction that for any turn $\tau \in (\tau_{o}, \tau_{1})$, all the deviation clusters are of size smaller than $\alpha x$. Particularly, at turn $\tau_{1}-1$, all the deviation clusters are of size smaller than $\alpha x$. Yet, by \Cref{claim:deviation_clusters}, the existence of the deviation cluster $Y^{(\tau_{1})}$, at size $d$, implies the existence of a deviation cluster at turn $\tau_{1} - 1$, of size at least $\alpha x$, which leads to a contradiction.
\end{proof}

\begin{claim}
Let $C$ be a quantum circuit with order $J$ and depth $d$, and let $\mathcal{E} \sim \mathcal{N}_{\mathcal{E}}$, so the circuit $C[\mathcal{E}]$ is a random variable, and it is fan-bounded by $\Delta$. For a time $t \le 2d$ and an integer $x \in \mathbb{N}$, let $A$ be the event in which, until time $t$, there is a time $t^{\prime} \le t$ such that there is a deviation cluster of size greater than $x$. For any step $\tau \in J$, let $B_{\tau}$ be the event that step $\tau$ is the first step to have a deviation cluster of size in the range $(\alpha x, x)$. Then:
  \begin{equation*}
    \begin{split}
      \prb{A} \le 2wd\max_{\tau \in [2wd]}\prb{B_{\tau}}
    \end{split}
  \end{equation*}
\end{claim}
\begin{proof}
By \Cref{claim:inter}, we have that the occurrence of a deviation cluster of size greater than $x$ before time $t$ implies that there is a step $\tau \in J$, such that $\tau \le 2wt$, in which there is a deviation cluster of size greater than $\alpha x$. Thus,
  \begin{equation*}
    \begin{split}
      A \subset \bigcup_{\tau \in [2wt]} B_{\tau} \subset \bigcup_{\tau \in [2wd]} B_{\tau}
    \end{split}
  \end{equation*}
  So by using the union bound we get:
  \begin{equation*}
    \begin{split}
      \prb{A}  \le \prb{\bigcup_{\tau \in [2wd]} B_{\tau} } & \le \sum_{\tau \in [2wd]}\prb{ B_{\tau} } \\
      &\le 2wd\max_{\tau \in [2wd]}\prb{B_{\tau}}
    \end{split}
  \end{equation*}
\end{proof}
