Informally, a quantum circuit is a placement of a collection of quantum gates, taken from a finite set, in different space-time locations. Each space-time location, associated with a gate, encodes the time of the application and the qubits in the support of the gate. Formally
\begin{definition}[Quantum Circuit]
  A space-time location $l$ is a tuple that encodes in its first coordinate time, and in its second coordinate a subset of integers referring to a support of (q)bits. 
  For a collection of quantum gates $\mathcal{G} = \{ g \colon L(\mathcal{H}_{2}^{\otimes \Delta} ) \rightarrow L(\mathcal{H}_{2}^{\otimes \Delta} ) \}$, and integers $w,d \in \mathbb{N}$ a quantum circuit $C$, with depth $d$ and width $w$, is a collection of tuples $\{ u_{i} = (g_{i},l_{i}) \}$, where $u_{i}$ are the gates of $C$ and each $u_{i}$ is composed of a gate $g_{i} \in \mathcal{G}$ and a space-time location $l_{i} \in \mathbb{Z}_{d} \times \mathbb{Z}_{w}^{\Delta}$. The gates of $C$ have to satisfy the constraint that for every two gates with the same time coordinate, there cannot be a non-empty intersection in their space coordinate, namely for $u_{i}=(g_{i},l_{i}),u_{j}=(g_{j},l_{j}) \in C$, the equality $l_{i,1} = l_{j,1}$ implies $l_{i,2} \cap l_{j,2} = \emptyset$. 
  We define the $t$-cut of $C$ to be $C|_{t} \colonequals \{ u_{i} = (g_{i},l_{i}) \colon u_{i} \in C \text{ and } l_{i,1} = t \}$.
\end{definition}
To define the running of a quantum circuit $C$ on a given (mixed) quantum state, we will need to have the notion of the acyclic directed graph associated with $C$.

\begin{definition}[Associated Computation DAG]
Let $C = \{u_{i}\}$ be a quantum circuit with depth $d$ and width $w$. We define the associated computation DAG of $C$, denoted by $\pi_{C}$, to be the acyclic directed graph $\pi_{C} = (\mathcal{U}, E)$ as follows:
  \begin{enumerate}
    \item The vertex set $\mathcal{U}$ is the gate set of $C$, namely $\mathcal{U} = \{u_{i}\}$.
    \item For $u_{i}= (g_{i},l_{i}), u_{j}= (g_{j},l_{j})$, There is an edge $\{u_{i},u_{j}\} \in E$ if and only if:
      \begin{equation*}
        \begin{split}
          j \in \arg \min_{j} \{ l_{j,1} : l_{j,1} > l_{i,1} \text{ and } l_{j,2}\cap l_{i,2} \neq \emptyset  \}
        \end{split}
      \end{equation*}
Note that there might be multiple indices that get the minimum, so the out degree might be greater than one.
  \end{enumerate}
\end{definition}

\begin{definition}
Let $\rho$ be a density matrix over $w$ qubits, and let $C = \{u_{i}\}$ be a quantum circuit with depth $d$ and width $w$. For any density matrix $\sigma$, quantum gate $g \in \mathcal{G}$, and location $l$, the application of the gate-location tuple $(g,l)$ on $\sigma$, denoted by $(g,l) \circ \sigma$, is the application of $g$ on $\sigma$ where wires are ordered such that the first qubit in the input to $g$ is $l_{2,1}$, the second is $l_{2,2}$, and so on. The result of the application of $C$ on $\rho$, denoted by $C \circ \rho$, is given by iterative applications of the gates $\{u_{i}\}$ on $\rho$, according to a topological order induced by $\pi_{C}$.

We say that $\rho_{1}, \rho_{2}, \ldots, \rho_{k}$ is a computation prefix if $\rho_{1} = \rho$ and $\rho_{i+1} = u^\prime_{i} \circ \rho_{i}$, where $\{u^{\prime}_{i}\}$ is a prefix of a topological ordering of $\{u_{i}\}$. If $\{u^{\prime}_{i}\}$ contains all the gates in $\{u_{i}\}$, then we call $\rho_{1}, \rho_{2}, \ldots, \rho_{k}$ a running.
\end{definition}



\begin{definition}[Associated Computation DAG]
Let $C = \{ u_{i} \}$ be a quantum circuit with depth $d$ and width $w$. For a set of fault locations $\mathcal{E} \subset [d] \times [w]$, we define the quantum circuit $C[\mathcal{E}] = \{ u^{\prime}_{i} \}$ to be the result of the following map:
  \begin{equation*}
    \begin{split}
      u_{i}=(g_{i},(l_{i,1}, l_{i,2})) & \mapsto   
      \begin{cases}
        (g_{i},( \inb{2} l_{i,1}, l_{i,2})) & u_{i} \in C \\ 
        (g_{i},( \inb{2} l_{i,1} \inb{ - 1 }, l_{i,2})) & u_{i} \in \mathcal{E}
      \end{cases}
    \end{split}
  \end{equation*}
We will denote the associated computation DAG of $C$ affected by $\mathcal{E}$ as $\comDAG$.
\end{definition}
\begin{definition}[Faults Propagation to Time $t$]
  For a quantum circuit $C=\{u_{i}\}$, with depth $d$, and gate $u_{j}$ in $C$, a propagation of a gate $u_{j} \in C$ to time $t$ is the quantum circuit obtained by the follow process: 
  \begin{enumerate}
    \item First define $C_{0}$ to be the quantum circuit which is the result of the map: 
  \begin{equation*}
    \begin{split}
      u_{i}=(g_{i},(l_{i,1}, l_{i,2})) & \mapsto   
      \begin{cases}
        (g_{i},( l_{i,1}, l_{i,2})) & l_{i,1} \le l_{j,1}  \text{ and } u_{i} \neq u_{j} \\ 
        (g_{i},(  l_{i,1} \inb{ + 1 }, l_{i,2})) & \text{else}
      \end{cases}
    \end{split}
  \end{equation*}
\item Then, for $k \le t - l_{j,1}$, construct $C_{k}$ from $C_{k-1}$ as follows: Denote $\tau = k + l_{j,1}$, and look for an operator $X \colon L(\mathcal{H}_{2}^{w}) \rightarrow L(\mathcal{H}_{2}^{w})$, such that $C_{k-1}|_{\tau}C_{k-1}|_{\tau+1} = C_{k-1}|_{\tau+1}X$. Then define
  \begin{equation*}
    \begin{split}
      C_{k} = C_{k-1}|_{1},C_{k-1}|_{2},..,C_{k-1}|_{\tau-1},C_{k-1}|_{\tau+1},X,C_{k-1}|_{\tau+2},..,C_{k-1}|_{d+1}
    \end{split}
  \end{equation*}
\item At the final stage of the process, namely where $k = t - l_{j,1}$, we define $X$ to be the propagation of $u_{j}$ to time $t$.
  \end{enumerate}
  For a quantum circuit $C$ and a set of faults $\mathcal{E} = \{ e_{i} \}$, denote by $X_{e_{i}}^{t}$ the propagation of $e_{i} \in \mathcal{E}$ in $C[\mathcal{E}]$. Let $j_{1}, j_{2}, \ldots, j_{l}$ be the indices of the faults in $\mathcal{E}$ such that their time coordinate is lower than $t$, ordered according to the reverse topological order induced by $\pi_{\mathcal{E}}$. The propagated error to time $t$ is defined by
  \begin{equation*}
    \begin{split}
      X^{t} \colonequals \prod_{i} X_{e_{j_{i}}}^{t} 
    \end{split}
  \end{equation*}
\end{definition}

\begin{definition}[Deviation]
  \label{def:deviation}
  For a quantum circuit $C$, a set of faults $\mathcal{E}$, and time $t$, let $X^{t}$ be the propagated error at time $t$. We define the deviation of $C[\mathcal{E}]$ from $C$ at time $t$ by the set of indices on which $X^{t}$ acts non-trivially. 
\end{definition}
\begin{definition}
Using the same notations as in \Cref{def:deviation}, for a set of unitary operators $\mathcal{S} = \{s \colon \mathcal{H}_{2}^{w} \rightarrow \mathcal{H}_{2}^{w} \}$, we define the propagated error up to stabilizers in $\mathcal{S}$ as $X^{t}[S] \colonequals \min_{s \in \mathcal{S}} \{ w(sX) \}$. Then, we define the deviation of $C[\mathcal{E}]$ from $C$ at time $t$ up to stabilizers in $\mathcal{S}$ to be the subset of coordinates on which $X^{t}[S]$ acts non-trivially.
\end{definition}
