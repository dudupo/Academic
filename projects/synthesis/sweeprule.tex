\section{Span Expansion}

Define the potentials functions $L_{i} : \mathbb{R}^{d} \rightarrow \mathbb{R}$ as follow\footnote{Notice that we use opposite signs relative to Berman and Peter Gacs, yet we agree with Perskil.}: 
\begin{equation*}
  \begin{split}
L_{i}(\xb{}) =    \begin{cases}
  \xb{}_{i} & i < d + 1 \\ 
  - \braket{\mathbf{1}, \xb{}} & i = d+1
\end{cases}
  \end{split}
\end{equation*}
The \textbf{Size} of a subset $S$ is defined as:   
\begin{equation*}
  \begin{split}
  \mathbf{Size}(S) = \sum_{i}\max_{\xb{} \in S} L_{i}(\xb{})
  \end{split}
\end{equation*}
Numerically, we show that the span property holds for the checks. Namely, let $e$ be an unsatisfied check at time $t$, and let $H_{e}$ be the unsatisfied checks at time $t-1$. Then there is a set $H^{\prime}_{e} \subset H_{e}$ such that:
\begin{enumerate}
  \item For any $i < d + 1$ there exists $e^{\prime} \in H^{\prime}_{e}$ such $L_{i}(e^{\prime}) \ge L_{i}(e)$. 
  \item There exists $e^{\prime} \in H^{\prime}_{e}$ such that $L_{d+1}(e^{\prime}) \le L_{d+1}(e) + 1 $. 
\end{enumerate}

  \begin{definition}[History, slice.]

    $G_{u}$ is the subgraph of $\braket{V, \text{Arrows} }$ induced by $ \{ v~\in V~:~\Time{v}~\le~\Time{u}\}$. $\History{u}$ is the connecting connected component of $G_{u}$ containing $u$. A slice is an equivalence class of the relation $u \equiv v$ iff $\History{u} = \History{v}$. Notice that if $u$ and $v$ are in the same slice $K$, then $\Time{u} = \Time{v}$. Thus we can write $\Time{K}$ and $\History{K}$ without ambiguity. 
  \end{definition}

  \begin{claim} 

If for two different slices $K_{1}$ and $K_{2}$ we have that $\Time{K_1}) = \Time{K_2})$, then $\History{K_1})$ and $\History{K_2})$ are disjoint. 
  \end{claim}
  \begin{proof}
    Directly from definitions. 
  \end{proof}

  \begin{definition}[The graph of slice]

    For a slice $K$ we define the graph $G_{K} = \braket{V_{K}, E_{K}}$ as follows:
    \begin{enumerate}
      \item $V_{K} = \{ \text{slice } R : R \subset \History{K} \text{ and } \Time{R} = \Time{K} - 1 \}$
      \item $ \{ R, S \} \in E_{K} $ iff $R,S \in V_{K}$ and for some $\{v,w\} \in \text{Forks}$ we have $v\in R$ and $w \in S$.
    \end{enumerate}
    
  \end{definition}

  \begin{claim}
    $G_{K}$ is connected. 
  \end{claim}

  \begin{proof}
    Assume $R$ and $S$ are from $V_{K}$ and that $v \in R, w \in S$. Since $v,w \in \History{K} $, there exists a path from $v$ to $w$ in $\History{K}$ consisting of arrows. Select this path in such a way that it contains the minimal number of points $x$ such that $\Time{x} = \Time{K}$. By induction on the number $k$ of such points we prove that $R$ and $S$ are in the same connected component of $G_{K}$. 

    \begin{enumerate}
      \item Base. $k=0$, In this case, the path from $v$ to $w$ lies within $G_{v} (= G_{w})$. Hence $\History{v} = \History{w}$ which means that the slices $R$ and $S$ are the identical. 
      \item Induction step. There exists $y$ on the path from $v$ to $w$ such that $\Time{y} = \Time{K}$. Let $x$ be the point which precedes $y$ on the path, and $z$ the point that follows $y$. Since the values of $\mathbf{Time}$ at the two ends of an arrow differ by $1$, and since $\Time{y}$ is minimal, we know that: 
        \begin{enumerate}
          \item $\Excuse{y} = \braket{x,z}$ or $\Excuse{y} = \braket{z,x}$, thus $\{x,z\}$ is a fork. 
          \item The slices of $x$ and $z$, say $T$ and $U$, are both members of $V_{K}$. 
        \end{enumerate}
        By the induction hypothesis, $R$ and $T$ are in the same connected component of $G_{K}$, and so are $U$ and $S$. On the other hand, $\{T,U\} \in E_{K}$.
    \end{enumerate}
  \end{proof}

  \begin{claim}
    Let $\mathbf{S} = \{ S_{1}, S_{2} , .. , S_{k} \}$ where $S_{i} \subset \mathbb{R}^{d}$, introduce an edge between $S_{i}$ and $S_{j}$ iff $S_{i} \cap S_{j} \neq \emptyset$. Assume that $\mathbf{S}$ is connected. Then $\Size{\cup^{k}_{i=1} S_{i}} \le \sum^{k}_{i=1}{\Size{S_{i}}}$.

        \Size{R_{1} \cap R_{2}} &= \sum_{i}^{d+1}{\max{ a_{i,1},a_{i,2} } } =  \sum_{i}^{d+1}{a_{i,1} + a_{i,2} - \min{ a_{i,1},a_{i,2} } } \\

    \end{equation*}
  \end{proof}

    \begin{enumerate}
      \item $W \cup \{v_{1} ,.., v_{d+1}\} \subset U \subset \History{K}$ and the graph $\braket{U, R\cup F}$ is connected. 
      \item All elements of $W$ are errors. 
      \item For $m = |W|$ and $s=\Span{K}$ we have $|R| \le \alpha (m-1) - \beta s $ and $|F| < m -1 $. 
    \end{enumerate}
  \end{definition}

  \begin{claim}
    Every spanned slice $K = \braket{K,v_{1},..,v_{d+1}}$ has an explanation.
  \end{claim}

          \item $\mathbf{R} = \{ \{v_{i}, \Excusei{v_{i}} \} \} \cup_{i=0}^{k}{ \mathbf{R}_{i} }$ with at most 
            \begin{equation*}
              \begin{split}
                & d+1 + \sum^{k}_{i=0}{\alpha (m_{i}-1)  - \beta s_{i} } \le d+1 + \alpha m - \alpha(k+1) - \beta\left(s - 1 + k \cdot \Span{\text{fork}}  \right) \\
                = & \alpha m - \beta s + \left( d+1 - (k+1) \alpha + \beta - \beta k \cdot \Span{\text{fork}} \right)
              \end{split}
            \end{equation*}
            So its enough to pick $\alpha, \beta$ such that for any $k \ge 1$: 
            \begin{equation*}
              \begin{split}
                & d+1 - (k+1) \alpha + \beta - \beta k f \le 0 \\ 
                \Rightarrow & \beta =1,  \alpha = (d+1)^{3}(1+f)  
              \end{split}
            \end{equation*}
        \end{enumerate}
        It is easy to see that $\mathbf{W}, \mathbf{R}, \mathbf{F}$ form an explanation of $K$. 
    \end{enumerate}
  \end{proof}
