
\section{Local Decoders}


\begin{definition}[DAG representation of quantum Gate]
Let $U$ be a quantum gate, composed of local gates $U_{1}, \ldots, U_{m}$, and a set of locations $V$. We define the directed graph $G_{U} = (V,E)$ such that the vertices are associated with the locations, and for locations $x,y \in V$, there is an edge $(x,y) \in E$ if and only if one of the following holds:
  \begin{enumerate}
    \item $x$ and $y$ correspond to two consecutive time steps of the same qubit. Namely, $y$ is the successor time location for $x$.
    \item There is a local gate $U_{i} \in U$ that acts at locations $x$ and $y$.
  \end{enumerate} 
\end{definition}

\begin{definition}[Light Cone] Let $U$ be a quantum gate, and let $q$ be a qubit on $U$ output corresponds to the output location $x_{q}$. We say that a location $x$ of $U$ \textbf{affects} $q$ if in $G_{U}$ there is a directed path between $x$ to $q$.  

  The \textbf{light cone} of $q$ respecting to $U$ is the union of all the locations of $U$ that affect $q$. We denote it by $U_{[q]}$ When $U$ is clear from the context, we will refer to the light cone respecting to it just as the \textbf{light cone of $q$}.
  
  \ctt{Grammar}
\end{definition}

\begin{definition}[Light Cones Intersections] 
  Let $U$ be a quantum gate, and let $q_{1}, q_{2}$ be two qubits on its output. We define the \textbf{light cones intersections} of $q_{1}$ and $q_{2}$ to be all the locations which affect both $q_{1}$ and $q_{2}$. We denote it by $U_{[q_{1},q_{2}]}$.  
  
  \ctt{Grammar}
\end{definition}


\begin{definition}[$U$-configurations]
  \label{def:config}
  Let $U$ be a quantum gate (not necessary unitary) , $x_{1}$ a location of $U$, and $X_1$ a single-qubit superoperator. Define $U[(x_1,X_1)]$ to be the quantum gate obtained from $U$ by placing $X_1$ at location $x_1$ as follows: 
  \begin{enumerate}
    \item If $x_1$ is a wait location, then we set $X_1$ to act at location $x_1$ instead. 
    \item If $x_1$ is a gate location, then we replace the gate with the gate which first applies $X_1$ and then applies the original gate.  
  \end{enumerate}
  For a set $X = \{ (x_{i},X_{i}) \}$ such that the $x_{i}$'s are locations and the $X_{i}$'s are single qubits superoperators. We denote by  $U[X]$ the $X$-configuration of $U$ defined to by placing each $X_{i}$ at the $x_{i}$ location. 
  \ctt{Grammar}
\end{definition}

\begin{definition}[Propgation]
  Let $U, X$ as in \Cref{def:config}, and let $Y=\{(y_{i},Y_i)\}$ be a set of pairs composed by locations and superoperators. We say that $U[Y]$ is a propagation of $U[X]$ if $U[X] = U[Y]$.  
\end{definition}


\begin{definition}[0/1-action of a Noise Channel]
  A noise channel $\mathcal{N} : L(\mathcal{H}) \rightarrow L(\mathcal{H})$ is superoperator that maps quantum circuit $U$ to a distribution over its possible configurations.  

  Fix an outcome of that distribution, namely a configuration $U[X]$, then: 
  \begin{enumerate}
      \item We define the $0$-action of $\mathcal{N}$ to be all locations in $U[X]$ that deviate from the locations on $U$, which are all the locations in $X$.  
      \item On the other hand, we define the $1$-action of $\mathcal{N}$ to be a set of locations, such for each location there exists at least one configuration $Y = \{(y_{i},Y_{i})\}$ such that $Y$ contains it and $U[Y]$ is a propagation of $U[X]$.
  \end{enumerate}

  \ctt{Grammar}
\end{definition}

\begin{definition}[0-Clean/Dirty Locations]
  \label{def:clean_dirty}
  Let $S$ be a subset of locations, and let $\mathcal{N}$ a noise channel acting on the quantum circuit. We define the event \textbf{$S$ is $0$-clean} to refer to all configurations in which no location in $S$ belongs to the $0$-action of  $\mathcal{N}$.

  Similarly, We define the event \textbf{ $S$ is $0$-dirty } to refer to all configurations for which at least one qubit in $S$ belongs to the $0$-action of $\mathcal{N}$.  

  When, $S$ is single a location associated with a qubit or a classical bit $q$, and the location is clear form the context, we will abuse the notation and use \textbf{$q$ is $0$-clean/dirty} to refer the event describing the whether or not the location is in the $0$-action.

  In addition, we will use the notations to refer events describing light cones, for example, for a quantum gate $U$ and two qubits $q_{1}$ and $q_{2}$ the event \textbf{Their light cones intersections is $0$-clean} refer to all configurations in which the subset of location in their intersection is $0$-clean. 
  
  \ctt{Grammar}
\end{definition}

\begin{remark}
  We add the digit $0$ at the beginning of the status to emphasis that the notation describes the influence of the channel independent to the computation defines by $U$. In particular, if location $y$ is in the light cone of $x$, and $y$ is dirty, then it doesn't implies that $x$ is also dirty.    
  \ctt{Grammar}
\end{remark}

\begin{definition}[1-Clean/Dirty Location]
  Using the same notations of \Cref{def:clean_dirty}. We say that a location $x$ is $1$-dirty if it belongs to the $1$-action of $\mathcal{N}$, and $1$-clean otherwise. 
  \ctt{Grammar}
\end{definition}



\begin{definition}[$p$-Local Stochastic Noise]
  Let $U$ be a quantum gate with a set of locations $\mathcal{L}$, we say that a channel $\mathcal{N}$ is a \textbf{local stochastic noise} channel if there exist a constant $p \in (0,1)$ such that $\mathcal{N}(U)$ defines a probability space in which for any $S \subset \mathcal{L}$, the probability that all the qubits in $S$ are $0$-dirty is upper bounded by $p^{|S|}$.
  \ctt{Grammar}
\end{definition}


\begin{definition}[$p$-Depolarized Noise]
  Let $U$ be a quantum gate with a set of locations $\mathcal{L}$, for a constant $p\in(0,1)$ we say that a channel $\mathcal{N}$ is a \textbf{$p$-depolarized noise} channel, if the event that a location $x$ is $0$-dirty is distributed by \textbf{Ber}$(p)$. In particular, the location are distributed independently.   
  \ctt{Grammar}
\end{definition}


\begin{definition} 
  Let $C$ be a quantum error correction code. We say that $D$ is a local decoder for $C$ if when applied against $2p(1-p)$-depolarized channel (local stochastic), the following holds:
  \begin{enumerate}
    \item \textbf{Local improvement.} There exists function $M : (0,1) \rightarrow (0,1) $ such that the probability of each of the output location being $1$-dirty is $M(2p(1-p)) < p^{\frac{100}{99}}$.
    \item \textbf{Locality computation.} There is a constant $l$ such that for any qubit $q$, there are  $ |D_{[q]}| \le l $. 
    \item \textbf{$0$-clean intersection implies 'independence'. } For any two qubits $q_{1},q_{2}$ on $D$ output, if their light cones intersection $D_{[q_{1},q_{2}]}$ is $0$-clean. Then:  
      \begin{equation*}
        \begin{split}
          \prb{ \text{both } q_{1},q_{2} \text{ are } 1\text{-dirty}   |  D_{[q_{1},q_{2}]} \text{ is } 0\text{-clean}} \le \prb{ q_{1} \text{ is } 1 \text{-dirty} } \cdot  \prb{ q_{2} \text{ is } 1 \text{-dirty} } 
        \end{split}
      \end{equation*}
  \end{enumerate}
For example, if we take $k$ large enough and consider the $(n,k)$-Toric, then both the swift-rule and the flip upon majority are local decoders.
\end{definition}


  \begin{definition}
Consider the moment after the application of the depolarizing channel, and right before the correction by the decoder.

For a subset of qubits $S$, we say that a qubit $x \in S$ is depended if there exists $y \in S$ such:
\begin{enumerate}
  \item Either $x$ and $y$ share a check that is not satisfied. (Type I)  
  \item Or, $x$ and $y$ have checks which share a bit that was flipped by the application of the channel. (Type II)
\end{enumerate}
  \end{definition}
